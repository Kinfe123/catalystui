### Example

```tsx
import { Tabs } from '@i4o-oss/catalystui'

export default () => {
	return (
		<Tabs
			defaultValue='1'
			tabs={[
				{ id: '1', title: 'Tab 1', content: 'Tab 1 Content' },
				{ id: '2', title: 'Tab 2', content: 'Tab 2 Content' },
				{ id: '3', title: 'Tab 3', content: 'Tab 3 Content' },
			]}
		/>
	)
}
```

### Props

| Name         | Type                                                                              | Default | Description                             |
| :----------- | :-------------------------------------------------------------------------------- | :------ | :-------------------------------------- |
| defaultValue | `string`                                                                          | `''`    | Id of the tab to be selected by default |
| tabs         | `Array<{ id: string, title: string \| ReactNode, content: string \| ReactNode }>` | []      | Array of tabs to render                 |
