### Example

```tsx
import { Checkbox } from '@i4o-oss/catalystui'

export default () => {
	return (
		<Checkbox
			defaultChecked={true}
			label={<p className='text-gray-900'>Checkbox</p>}
		/>
	)
}
```

### Props

| Name             | Type                  | Default     | Description                                |
| :--------------- | :-------------------- | :---------- | :----------------------------------------- |
| `defaultChecked` | `boolean`             | `false`     | Whether the checkbox is checked by default |
| `disabled`       | `boolean`             | `false`     | Whether the checkbox is disabled           |
| `label`          | `string \| ReactNode` | `undefined` | The label of the checkbox                  |
