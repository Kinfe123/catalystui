import { Nav } from '@i4o-oss/catalystui'

	<Nav
		items={[
			{
				id: 'home',
				label: 'Home',
				href: '/',
			},
			{
				id: 'about',
				label: 'About',
				href: '/about',
			},
			{
				id: 'contact',
				label: 'Contact',

### Props

| Name  | Type            | Default | Description               |
| :---- | :-------------- | :------ | :------------------------ |
| items | Array           | []      | Array of navigation items |
| type  | `row \| column` | 'row'   | Type of navigation menu   |
